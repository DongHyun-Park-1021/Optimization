// f(x1, ... xn) = (log( sum_{i=1}^{m} exp(a1i x1 + ... + ani xn) ) ) / (sum_{i=1}^{m} 1/(1 + bi exp(xi)) )
double f(vector<double> x) {
    double dom = 0, temp;
    for (int i = 0 ; i < m; i++) {
        temp = 0;
        for (int j = 0; j < dim; j++) {
            temp += (((i+1) * (j+1)) * x[j] / ((i + j + 1) * (i + j + 1)));
        }
        temp += (1/(i+1));
        dom += exp(temp);
    }
    dom = log(dom);

    double quotient = 0;
    for (int i = 0 ; i < dim ; i++) {
        quotient += 1/(1 + (1/(i+1)) * exp(x[i]));
    }
    return dom / quotient;
}

// g(x1, ... xn) = x1^2 + ... xn^2 - 1
double g(vector<double> x) {
    double ret = 0;
    for (int i = 0 ; i < dim ; i++) {
        ret += (x[i] * x[i]);
    }
    return ret - 1;
}
